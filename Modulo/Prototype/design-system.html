Usage in Prototypes

HTML (add script tag + use data-lucide)
<!-- Add before </body> -->
<script src="https://unpkg.com/lucide@latest"></script>
<script>lucide.createIcons();</script>

<!-- Use anywhere -->
<i data-lucide="arrow-right"></i>
<i data-lucide="wallet" style="--bk-icon-size:24px;"></i>
<i data-lucide="settings" style="color:var(--bk-brand-primary);"></i>

<!-- Override stroke weight -->
<i data-lucide="bell" style="--bk-icon-stroke:1;"></i>

Token Integration

/* Charts read BakeKit tokens via JS */
const styles = getComputedStyle(document.documentElement);
const brandColor = styles.getPropertyValue('--bk-brand-primary').trim();
const successColor = styles.getPropertyValue('--bk-success').trim();
const errorColor = styles.getPropertyValue('--bk-error').trim();
const bgColor = styles.getPropertyValue('--bk-bg-base').trim();
const textColor = styles.getPropertyValue('--bk-text-muted').trim();

/* Lightweight Charts */
chart.applyOptions({
  layout: { background: { color: bgColor }, textColor },
});
